cmake_minimum_required(VERSION 3.10)

project(MapReduce VERSION 1.0.0 LANGUAGES CXX)

# Устанавливаем стандарт C++17
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# Опции сборки
option(BUILD_TESTS "Build unit tests" OFF)

file(COPY ${CMAKE_SOURCE_DIR}/input/ DESTINATION ${CMAKE_BINARY_DIR}/input/)

# Основной исполняемый файл
add_executable(mapreduce 
    src/main.cpp
    src/mapper.cpp
    src/reducer.cpp
    src/statistics.cpp
)

target_include_directories(mapreduce PRIVATE ${CMAKE_SOURCE_DIR}/src)

# Устанавливаем флаги компиляции
target_compile_options(mapreduce PRIVATE 
    -Wall 
    -Wextra 
    -O2
)

# Если нужны отдельные исполняемые файлы для маппера и редуктора
add_executable(mapper src/mapper.cpp src/statistics.cpp)
add_executable(reducer src/reducer.cpp src/statistics.cpp)

target_include_directories(mapper PRIVATE ${CMAKE_SOURCE_DIR}/src)
target_include_directories(reducer PRIVATE ${CMAKE_SOURCE_DIR}/src)

# Установка в bin директорию
install(TARGETS mapreduce mapper reducer
    RUNTIME DESTINATION bin
)

# Тесты (опционально)
if(BUILD_TESTS)
    enable_testing()
    add_executable(tests 
        tests/test_statistics.cpp
        src/statistics.cpp
    )
    target_include_directories(tests PRIVATE ${CMAKE_SOURCE_DIR}/src)
    target_link_libraries(tests PRIVATE gtest_main)
    include(GoogleTest)
    gtest_discover_tests(tests)
endif()
